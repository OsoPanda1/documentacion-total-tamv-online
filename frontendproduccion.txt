// Root Layout Component
function RootLayout() {
  const { isAuthenticated, loading: authLoading } = useAuth()
  const { startMonitoring, recordMetric } = usePerformance()
  const [federationReady, setFederationReady] = useState(false)
  const [isabellaReady, setIsabellaReady] = useState(false)

  useEffect(() => {
    // Inicializar monitoring de performance
    startMonitoring()

    // Inicializar sistemas principales
    const initializeSystems = async () => {
      try {
        // 1. Validar y registrar federación triple
        const federationResult = await FederationAudit.validateAll()
        recordMetric('federation_validation', federationResult.status === 'ok' ? 1 : 0)
        setFederationReady(true)

        // 2. Inicializar Isabella Core Protocol
        if (isAuthenticated) {
          await initializeIsabella({
            userId: authStore.userId,
            channel: 'PUBLICUI',
          })
          setIsabellaReady(true)
        }

        // 3. Inicializar Quantum Filter
        await initializeQuantumFilter({
          enableQuantumMode: true,
          performanceLevel: 'production',
        })

        recordMetric('system_initialization', 1)
      } catch (error) {
        console.error('Error initializing systems:', error)
        recordMetric('system_initialization_error', 1)
      }
    }

    if (!authLoading) {
      initializeSystems()
    }
  }, [authLoading, isAuthenticated])

  if (authLoading || !federationReady) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-black">
        <LoadingScreen stage={authLoading ? 'auth' : 'federation'} />
      </div>
    )
  }

  return (
    <ErrorBoundary>
      <QueryClientProvider client={queryClient}>
        <Suspense fallback={<LoadingScreen stage="router" />}>
          <div className="tamv-root-container">
            <RouterProvider router={router} />
            <Toaster position="bottom-right" theme="dark" />
          </div>
        </Suspense>
      </QueryClientProvider>
    </ErrorBoundary>
  )
}

// LoadingScreen Component
function LoadingScreen({ stage }: { stage: 'auth' | 'federation' | 'router' }) {
  const messages = {
    auth: 'Authenticating...',
    federation: 'Validating triple federation...',
    router: 'Loading application...',
  }

  return (
    <div className="loader-container">
      <div className="matrix-loader">
        {[...Array(20)].map((_, i) => (
          <div key={i} className="matrix-line" style={{ animationDelay: `${i * 0.1}s` }} />
        ))}
      </div>
      <p className="loader-text text-glow">{messages[stage]}</p>
    </div>
  )
}

export default function AppWrapper() {
  return <RootLayout />
}

import React, { useState, useCallback, useEffect } from 'react'
import { useLocation } from '@tanstack/react-router'
import { useUIStore } from '@/core/state/uiStore'
import { useNotifications } from '@/core/hooks/useNotifications'
import { NavBar } from './NavBar'

import { NotificationIndicator } from './NotificationIndicator'
import './MainLayout.css'

interface BarConfig {
  id: 'top' | 'left' | 'right' | 'bottom'
  isCollapsed: boolean
  hasNotifications: boolean
  notificationColor: string
  notificationPulse: boolean
}

export function MainLayout({ children }: { children: React.ReactNode }) {
  const location = useLocation()
  const { barStates, toggleBar, expandBar, collapseBar } = useUIStore()
  const { unreadCount, categorizedNotifications } = useNotifications()
  
  const [barConfig, setBarConfig] = useState<Record<string, BarConfig>>({
    top: { id: 'top', isCollapsed: false, hasNotifications: false, notificationColor: '', notificationPulse: false },
    left: { id: 'left', isCollapsed: true, hasNotifications: false, notificationColor: '', notificationPulse: false },
    right: { id: 'right', isCollapsed: true, hasNotifications: false, notificationColor: '', notificationPulse: false },
    bottom: { id: 'bottom', isCollapsed: true, hasNotifications: false, notificationColor: '', notificationPulse: false },
  })

  // Determinar qué notificaciones mostrar en cada barra
  useEffect(() => {
    const newConfig = { ...barConfig }

    // Actualizar indicadores de notificaciones
    if (categorizedNotifications.messages?.length > 0) {
      newConfig.left.hasNotifications = true
      newConfig.left.notificationColor = '#3B82F6' // Azul para mensajes
    }

    if (categorizedNotifications.payments?.length > 0) {
      newConfig.right.hasNotifications = true
      newConfig.right.notificationColor = '#10B981' // Verde para pagos
    }

    if (categorizedNotifications.invitations?.length > 0) {
      newConfig.left.hasNotifications = true
      newConfig.left.notificationColor = '#A78BFA' // Púrpura para invitaciones
    }

    if (unreadCount > 0) {
      newConfig.top.hasNotifications = true
      newConfig.top.notificationColor = '#F59E0B' // Ámbar para general
      newConfig.top.notificationPulse = true
    }

    setBarConfig(newConfig)
  }, [categorizedNotifications, unreadCount])

  // Sistema de expandir una barra automáticamente cuando se recibe notificación
  const handleNotificationIncoming = useCallback((notification: any) => {
    switch (notification.type) {
      case 'MESSAGE':
        expandBar('left')
        break
      case 'PAYMENT':
        expandBar('right')
        break
      case 'INVITATION':
        expandBar('left')
        break
      case 'SYSTEM':
        expandBar('top')
        break
    }
  }, [expandBar])

  // Determinar visibilidad de barras según contexto de usuario
  const shouldShowLeftBar = location.pathname !== '/settings'
  const shouldShowRightBar = location.pathname !== '/settings'
  const shouldShowBottomBar = !location.pathname.includes('/xr/')

  return (
    <div className="tamv-main-layout">
      {/* Contenedor de barras con titileo de notificaciones */}
      <div className="bars-container">
        {/* BARRA SUPERIOR - Módulos principales y buscador global */}
        <NavBar
          isCollapsed={barConfig.top.isCollapsed}
          onToggle={() => toggleBar('top')}
          hasNotification={barConfig.top.hasNotifications}
          notificationColor={barConfig.top.notificationColor}
          pulse={barConfig.top.notificationPulse}
        />

        {/* Contenedor principal de contenido */}
        <div className="main-content-wrapper">
          {/* BARRA LATERAL IZQUIERDA - Módulos sociales del creador */}
          {shouldShowLeftBar && (
            <SidebarLeft
              isCollapsed={barConfig.left.isCollapsed}
              onToggle={() => toggleBar('left')}
              hasNotification={barConfig.left.hasNotifications}
              notificationColor={barConfig.left.notificationColor}
              pulse={barConfig.left.notificationPulse}
            />
          )}

          {/* Contenido principal */}
          <main className="content-area">

  private analyzeEmotionQuantum(message: string): {
    primary: string;
    intensity: number;
    valence: number;
  } {

      joy: ["feliz", "alegre", "excelente", "maravilloso", "increíble"],
      sadness: ["triste", "mal", "deprimido", "solo", "perdido"],

      "anticipation",

      primary: maxEmotion,

    const dataBuffer = encoder.encode(data);
    const hashBuffer = await crypto.subtle.digest("SHA-256", dataBuffer);
    const hashArray = Array.from(new Uint8Array(hashBuffer));
    return hashArray.map((b) => b.toString(16).padStart(2, "0")).join("");

  /**
   * Procesamiento principal de intents de Isabella
   * Flujo: Validación → Análisis emocional → Guardianes → Generación → BlockchainRegistry
   */
  async processIntent(intent: IsabellaIntent): Promise<IsabellaResponse> {
    const startTime = Date.now();
    const requestId = intent.requestId || crypto.randomUUID();

    try {
      this.logger.log(`[${requestId}] Procesando intent de ${intent.userId}`);

      // PASO 1: Validación de entrada
      this.validateIntent(intent);

      // PASO 2: Análisis emocional cuántico
      const emotionalAnalysis = await this.analyzeEmotionQuantum(
        intent.message,
        intent.emotionalState
      );

      // PASO 3: Consultar guardianes éticos
      const ethicalValidation = await this.guardianSystem.validate({
        userId: intent.userId,
        message: intent.message,
        emotionalState: emotionalAnalysis,
        sourceChannel: intent.sourceChannel,
      });

      if (!ethicalValidation.approved) {
        return this.createBlockedResponse(
          requestId,
          intent.userId,
          ethicalValidation.reason,
          emotionalAnalysis,
          startTime
        );
      }

      // PASO 4: Absorber contexto de internet si es necesario
      let enrichedContext = intent.context || {};
      if (this.shouldEnrichWithWebContent(intent.message)) {
        enrichedContext = await this.knowledgeEngine.absorb(intent.message);
      }

      // PASO 5: Generar respuesta inteligente
      const responseContent = await this.generateIntelligentResponse(
        intent.message,
        emotionalAnalysis,
        enrichedContext
      );

      // PASO 6: Registrar en blockchain (BookPI)
      const blockchainHash = await this.blockchainConnector.recordEvent({
        requestId,
        userId: intent.userId,
        message: intent.message,
        response: responseContent,
        emotionalSignature: emotionalAnalysis,
        timestamp: new Date().toISOString(),
      });

      // PASO 7: Actualizar memoria emocional
      await this.emotionalMemory.update(intent.userId, emotionalAnalysis);

      // PASO 8: Persistir en Supabase
      await this.persistInteraction(requestId, intent, responseContent);

      // PASO 9: Construir respuesta final
      const response: IsabellaResponse = {
        id: requestId,
        userId: intent.userId,
        content: responseContent,
        emotionalSignature: emotionalAnalysis,
        blockchainHash,
        confidenceScore: this.calculateConfidence(
          intent.message,
          responseContent
        ),
        processingTimeMs: Date.now() - startTime,
        securityLevel: ethicalValidation.securityLevel,
      };

      this.logger.log(
        `[${requestId}] Intent procesado exitosamente en ${response.processingTimeMs}ms`
      );
      return response;
    } catch (error) {
      this.logger.error(
        `[${requestId}] Error procesando intent: ${error.message}`
      );
      throw error;
    }
  }

  /**
   * Análisis emocional con inspiración cuántica
   * Superposición de estados emocionales + colapso a respuesta
   */
  private async analyzeEmotionQuantum(
    message: string,
    userState?: any
  ): Promise<EmotionalSignature> {
    const emotionKeywords: Record<string, string[]> = {
      joy: ["feliz", "alegre", "excelente", "maravilloso", "increíble"],
      sadness: ["triste", "mal", "deprimido", "solo", "perdido"],
      anger: ["enojado", "furioso", "molesto", "irritado"],
      fear: ["miedo", "asustado", "nervioso", "ansioso"],
      trust: ["confo", "seguro", "creo", "apoyo"],
      surprise: ["sorprendido", "wow", "asombroso"],
      disgust: ["asco", "horrible", "desagradable"],
      anticipation: ["espero", "emocionado", "pronto"],
    };

    const lowerMsg = message.toLowerCase();
    let maxEmotion = "neutral";
    let maxScore = 0;

    for (const [emotion, keywords] of Object.entries(emotionKeywords)) {
      const score = keywords.filter((k) => lowerMsg.includes(k)).length;
      if (score > maxScore) {
        maxScore = score;
        maxEmotion = emotion;
      }
    }

    // Superposición cuántica: combinar análisis de palabras + contexto
    const wordEmotionScore = Math.min(maxScore / 3, 1.0);
    const contextBoost = userState?.intensity || 0.5;
    const finalIntensity = (wordEmotionScore + contextBoost) / 2;

    // Colapso a estado emocional
    const valence = ["joy", "trust", "anticipation"].includes(maxEmotion)
      ? 0.7
      : ["sadness", "fear", "disgust"].includes(maxEmotion)
        ? -0.7
        : 0;

    const arousal = ["anger", "surprise", "joy"].includes(maxEmotion) ? 0.8 : 0.4;

    const emotionalSig: EmotionalSignature = {
      primary: maxEmotion,
      intensity: finalIntensity,
      valence,
      arousal,
      coherenceLevel: this.coherenceLevel,
    };

    // Registrar para auditoría
    await this.logEmotionalAnalysis(emotionalSig);

    return emotionalSig;
  }
